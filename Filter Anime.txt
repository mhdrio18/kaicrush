document.addEventListener('click', function (e) {
  if (e.target && e.target.id === 'applyFilter') {
    const modal = document.querySelector('.ga-modal');
    
    // Mengambil genre yang dipilih
    const selectedGenres = Array.from(modal.querySelectorAll('.filter-choice.active'))
      .map(el => el.textContent.trim());
    
    // Mengambil type yang dipilih
    const selectedType = modal.querySelector('.filter-block:nth-child(1) .dropdown-item.active')?.textContent.trim();
    
    // Mengambil status yang dipilih
    const selectedStatus = modal.querySelector('.filter-block:nth-child(2) .dropdown-item.active')?.textContent.trim();
    
    // Mengambil season yang dipilih
    const selectedSeason = modal.querySelector('.filter-block:nth-child(3) .dropdown-item.active')?.textContent.trim();
    
    // Mengambil sort yang dipilih
    const selectedSort = modal.querySelector('.filter-block:nth-child(4) .dropdown-item.active')?.textContent.trim();

    // Konversi nilai sort ke kunci filter
    let orderFilter = "published";
    if (selectedSort === "Recently Updated") orderFilter = "updated";
    else if (selectedSort === "Recently Added") orderFilter = "published";
    else if (selectedSort === "Name A-Z") orderFilter = "title";
    else if (selectedSort === "Released Date") orderFilter = "published";

    // Hapus modal setelah apply
    modal.remove();
    document.body.style.overflow = '';
document.body.style.paddingRight = '';

    // Menampilkan filter yang dipilih pada .selected-list
    const selectedItems = [];

    // Menambahkan genre yang dipilih
    selectedGenres.forEach(genre => {
      selectedItems.push(`<a class="item"><div class="item-flex"><span>${genre}</span></div></a>`);
    });

    // Menambahkan type yang dipilih
    if (selectedType && selectedType !== 'All') {
      selectedItems.push(`<a class="item"><div class="item-flex"><span>${selectedType}</span></div></a>`);
    }

    // Menambahkan status yang dipilih
    if (selectedStatus && selectedStatus !== 'All') {
      selectedItems.push(`<a class="item"><div class="item-flex"><span>${selectedStatus}</span></div></a>`);
    }

    // Menambahkan season yang dipilih
    if (selectedSeason && selectedSeason !== 'All') {
      selectedItems.push(`<a class="item"><div class="item-flex"><span>${selectedSeason}</span></div></a>`);
    }

    // Menambahkan sort yang dipilih
    if (selectedSort && selectedSort !== 'All') {
      selectedItems.push(`<a class="item"><div class="item-flex"><span>${selectedSort}</span></div></a>`);
    }

    // Menampilkan hasil di .selected-list
    document.querySelector('.selected-list').innerHTML = selectedItems.join('');

    // Panggil fungsi fetchAndProcessData dengan parameter yang sesuai
    fetchAndProcessData(
      selectedSeason !== 'All' ? [selectedSeason] : [],
      selectedStatus !== 'All' ? [selectedStatus] : [],
      selectedType !== 'All' ? [selectedType] : [],
      selectedGenres,
      orderFilter
    );
  }
});
